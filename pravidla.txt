PROGRAM -> function DEF_FUNC

DEF_FUNC -> main ( ) STAT end ;
DEF_FUNC -> idFunc ( PARAMS ) STAT end PROGRAM

PARAMS -> eps
PARAMS -> id NEXT_PARAMS
NEXT_PARAMS -> eps
NEXT_PARAMS -> , id NEXT_PARAMS

STAT -> eps
STAT -> DEF_VAR STAT_LIST

DEF_VAR -> local id ; DEF_VAR_NEXT
DEF_VAR_NEXT -> eps
DEF_VAR_NEXT -> DEF_VAR

STAT_LIST -> eps
STAT_LIST -> if     expression then STAT_LIST else STAT_LIST end ; STAT_LIST
STAT_LIST -> while  expression then STAT_LIST end ; STAT_LIST
STAT_LIST -> return expression ;
STAT_LIST -> id =   EXP ; STAT_LIST

EXP -> expression
EXP -> idFunc ( VAR_PARAMS )

VAR_PARAMS -> VAR VAR_P_NEXT
VAR -> string
VAR -> nil
VAR -> number
VAR -> id
VAR -> true
VAR -> false
VAR_P_NEXT -> eps
VAR_P_NEXT -> , VAR VAR_P_NEXT

function idFunc ( id , id , id ) 
  local id ; 
  id = expression ; 
  if expression then 
    id = idFunc ( number , string , nil , true , false ) ; 
  else 
  end ; 
end
